Select nc.NUM_CLASS, nc.LETTER, ns.NAME_SUBJECT, nle.TIME_LESSON, wd.WEEK_DAY
FROM SCHEDULES sc
LEFT JOIN NUM_CLASSES nc ON nc.ID_NUM_CLASS = sc.ID_NUM_CLASS
LEFT JOIN WEEK_DAYS wd ON wd.ID_WEEK_DAY = sc.ID_WEEK_DAY
LEFT JOIN NAME_SUBJECTS ns ON ns.ID_NAME_SUBJECT = sc.ID_NAME_SUBJECT
LEFT JOIN NUM_LESSONS nle ON nle.ID_NUM_LESSON = sc.ID_NUM_LESSON
WHERE nc.NUM_CLASS = 10
AND nc.LETTER ='A'
ORDER BY wd.ID_WEEK_DAY, nle.ID_NUM_LESSON
